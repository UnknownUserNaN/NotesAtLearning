给定两个整数数组 inorder 和 postorder ，其中 inorder 是二叉树的中序遍历， postorder 是同一棵树的后序遍历，请你构造并返回这颗 二叉树 。

提示：
- inorder 和 postorder 都由 不同 的值组成
- postorder 中每一个值都在 inorder 中

---
这题可以参考”105.从前序与中序遍历序列构造二叉树“的思路，使用递归的思路解决问题。容易知道，后序遍历的最后一个元素必然是根，随后便可以在中序遍历中找到左右两棵子树的长度。思路与之前是类似的，不多说了，直接看程序：
```cpp
/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode() : val(0), left(nullptr), right(nullptr) {}
 *     TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
 *     TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left),
 * right(right) {}
 * };
 */
class Solution {
    unordered_map<int, int> num2index;

public:
    TreeNode* recursiveBuild(vector<int>& in, vector<int>& post, int l1, int r1,
                             int l2, int r2) {
        // 取出后序列表中的最后一个元素作为根
        int root_val = post[r2];
        TreeNode* root = new TreeNode(root_val);

        // 在中序列表中找到根节点位置，进而计算出左右子树的结点数
        int root_idx = num2index[root_val];
        int n_left = root_idx - l1;
        int n_right = r1 - root_idx;

        // 递归建立左右子树
        if (n_left == 0)
            root->left = nullptr;
        else
            root->left =
                recursiveBuild(in, post, l1, root_idx - 1, l2, l2 + n_left - 1);
        if (n_right == 0)
            root->right = nullptr;
        else
            root->right = recursiveBuild(in, post, root_idx + 1, r1,
                                         r2 - n_right, r2 - 1);

        return root;
    }

    TreeNode* buildTree(vector<int>& inorder, vector<int>& postorder) {
        // 为搜索简便，建立一个哈希表
        for (int i = 0; i < inorder.size(); i++)
            num2index[inorder[i]] = i;

        // 调用递归方法建立树
        return recursiveBuild(inorder, postorder, 0, inorder.size() - 1, 0,
                              postorder.size() - 1);
    }
};
```
迭代法的程序与上面类似，时间复杂度和空间复杂度也与之大致相同。这里直接贴上[官方题解](https://leetcode.cn/problems/construct-binary-tree-from-inorder-and-postorder-traversal/solutions/426738/cong-zhong-xu-yu-hou-xu-bian-li-xu-lie-gou-zao-14/)的做法：
```cpp
class Solution {
public:
    TreeNode* buildTree(vector<int>& inorder, vector<int>& postorder) {
        if (postorder.size() == 0) {
            return nullptr;
        }
        auto root = new TreeNode(postorder[postorder.size() - 1]);
        auto s = stack<TreeNode*>();
        s.push(root);
        int inorderIndex = inorder.size() - 1;
        for (int i = int(postorder.size()) - 2; i >= 0; i--) {
            int postorderVal = postorder[i];
            auto node = s.top();
            if (node->val != inorder[inorderIndex]) {
                node->right = new TreeNode(postorderVal);
                s.push(node->right);
            } else {
                while (!s.empty() && s.top()->val == inorder[inorderIndex]) {
                    node = s.top();
                    s.pop();
                    inorderIndex--;
                }
                node->left = new TreeNode(postorderVal);
                s.push(node->left);
            }
        }
        return root;
    }
};
```
