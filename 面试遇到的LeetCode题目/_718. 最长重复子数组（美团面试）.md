给两个整数数组 nums1 和 nums2 ，返回 两个数组中 公共的 、长度最长的子数组的长度 。

---
本题需要用到多维的动态规划。

```cpp
class Solution {
public:
    int findLength(vector<int>& nums1, vector<int>& nums2) {
        int l1 = nums1.size(), l2 = nums2.size();

        vector<vector<int>> dp(l1 + 1, vector<int>(l2 + 1));
        int ans = 0;
        for (int i = 1; i <= l1; i++) {
            for (int j = 1; j <= l2; j++) {
                dp[i][j] = nums1[i-1] == nums2[j-1] ? dp[i - 1][j - 1] + 1 : 0;
                ans = max(ans, dp[i][j]);
            }
        }

        return ans;
    }
};
```

也可以优化成一维的滚动数组，可以降低空间复杂度：
```cpp
class Solution {
public:
    int findLength(vector<int>& nums1, vector<int>& nums2) {
        int l1 = nums1.size(), l2 = nums2.size();

        vector<int> dp(l2 + 1);
        int ans = 0;
        for (int i = 1; i <= l1; i++) {
            for (int j = l2; j >= 1; j--) {
                dp[j] = nums1[i-1] == nums2[j-1] ? dp[j - 1] + 1 : 0;
                ans = max(ans, dp[j]);
            }
        }

        return ans;
    }
};
```
