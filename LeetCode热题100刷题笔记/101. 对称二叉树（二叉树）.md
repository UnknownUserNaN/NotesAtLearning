给你一个二叉树的根节点 root ， 检查它是否轴对称。

进阶：你可以运用递归和迭代两种方法解决这个问题吗？

---
本题通过广度优先的思路实现非常简单，只需要在“扩展”完一层的二叉树之后，检查当前队列中的结点存储的值是否对称罢了。唯一的不足之处在于，程序在运行过程中需要考虑空结点。也就是说，将输入的整个二叉树看作一个完全二叉树，然后逐层遍历。
写出的程序如下。毫无意外地，这个算法超时了，因为这个做法的耗时实在太大。
```cpp
/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode() : val(0), left(nullptr), right(nullptr) {}
 *     TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
 *     TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left),
 * right(right) {}
 * };
 */
class Solution {
public:
    bool isSymmetric(TreeNode* root) {
        vector<TreeNode*> nodes;
        nodes.push_back(root);
        while (!nodes.empty()) {
            // 记录当前层的结构
            int n = nodes.size();
            for (int i = 0; i < n; i++) {
                if (nodes.front() == nullptr) {
                    // 注意，这里考虑了空指针，空指针也可以“分裂”
                    nodes.push_back(nullptr);
                    nodes.push_back(nullptr);
                } else {
                    nodes.push_back(nodes.front()->left);
                    nodes.push_back(nodes.front()->right);
                }
                nodes.erase(nodes.begin());
            }

            // 检查当前层是否全是空指针，若是则提前结束该循环
            bool flag = true;
            for (int i = 0; i < 2 * n; i++)
                if (nodes[i] != nullptr) {
                    flag = false;
                    break;
                }
            if (flag)
                break;

            // 检查当前层的对称性
            for (int i = 0; i < n; i++) {
                TreeNode *p1 = nodes[i], *p2 = nodes[2 * n - 1 - i];
                if (p1 == nullptr && p2 != nullptr)
                    return false;
                if (p1 != nullptr && p2 == nullptr)
                    return false;
                if (p1 != nullptr && p2 != nullptr && p1->val != p2->val)
                    return false;
            }
        }
        return true;
    }
};
```
没办法，只能换一个思路。题目中的进阶提示这个问题有递归和迭代两种方法。递归方法显然是更简单的。为递归函数传入两个结点left和right，如果left的左孩子与right的右孩子相同，left的右孩子与right的左孩子相同，并且这两个结点的值也相同，那么这两个结点也对称。基于这个思路，就可以写出程序：
```cpp
class Solution {
public:
    bool checkSymetric(TreeNode* p1, TreeNode* p2) {
        if (p1 == nullptr && p2 != nullptr)
            return false;
        if (p1 != nullptr && p2 == nullptr)
            return false;
        if (p1 == nullptr && p2 == nullptr)
            return true;
        if (checkSymetric(p1->left, p2->right) &&
            checkSymetric(p1->right, p2->left) && p1->val == p2->val)
            return true;
        else
            return false;
    }

    bool isSymmetric(TreeNode* root) {
        return checkSymetric(root->left, root->right);
    }
};
```
官方题解的写法更为简洁：
```cpp
class Solution {
public:
    bool check(TreeNode *p, TreeNode *q) {
        if (!p && !q) return true;
        if (!p || !q) return false;
        return p->val == q->val && check(p->left, q->right) && check(p->right, q->left);
    }

    bool isSymmetric(TreeNode* root) {
        return check(root->left, root->right);
    }
};
```
有了递归法的思路，迭代法（循环法）也就迎刃而解了。只需要借用栈或者队列的结构，重现上面的思路即可。对称地遍历两个根节点的子树，对比遍历过程中各结点的值，如遇到不同的值，则立即返回false。
这里直接在官方题解程序的基础上修改了：
```cpp
class Solution {
public:
    bool isSymmetric(TreeNode* root) {
        queue<TreeNode*> q;
        q.push(root->left);
        q.push(root->right);
        while (!q.empty()) {
            TreeNode* u = q.front();
            q.pop();
            TreeNode* v = q.front();
            q.pop();
            
            if (!u && !v)
                continue;
            if ((!u || !v) || (u->val != v->val))
                return false;

            q.push(u->left);
            q.push(v->right);

            q.push(u->right);
            q.push(v->left);
        }
        return true;
    }
};
```
实际上，上面的算法也可以使用栈来实现。这两种实现方式的差异在于，队列是类似于“广度优先“的遍历，而栈则是类似于”深度优先“的遍历。
```cpp
class Solution {
public:
    bool isSymmetric(TreeNode* root) {
        stack<TreeNode*> stk;
        stk.push(root->left);
        stk.push(root->right);
        
        while (!stk.empty()) {
            TreeNode* q = stk.top(); stk.pop();
            TreeNode* p = stk.top(); stk.pop();
            
            if (!p && !q) continue;
            if (!p || !q) return false;
            if (p->val != q->val) return false;
            
            stk.push(p->left);
            stk.push(q->right);
            
            stk.push(p->right);
            stk.push(q->left);
        }
        return true;
    }
};
```
