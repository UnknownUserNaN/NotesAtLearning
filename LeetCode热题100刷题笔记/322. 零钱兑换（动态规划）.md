给你一个整数数组 coins ，表示不同面额的硬币；以及一个整数 amount ，表示总金额。

计算并返回可以凑成总金额所需的 最少的硬币个数 。如果没有任何一种硬币组合能组成总金额，返回 -1 。

你可以认为每种硬币的数量是无限的。

---
本题与“279.完全平方数”很像。用动态规划数组表示我们最终希望的求得的解是`dp[amount]`，那么该状态可以由数组中的其他状态通过不同的硬币数“转移”而来。我们只需要在这些转移的状态中取一个最小值，于是此时该动态规划数组元素的解即为该最小值加一。
当然本题还需要处理一种额外情况，即没有任何一种硬币组合能组成总金额的情况。这实际上很容易做到，只要`amount<min(coins)`，就没有对应的解，我们便可以输出-1。
基于上面的思考很容易写出下面的程序：
```cpp
class Solution {
public:
    int coinChange(vector<int>& coins, int amount) {
        if (amount == 0)
            return 0;

        // 将硬币数组转为哈希表，方便查询，并记录面额最小的硬币
        unordered_map<int, bool> m;
        int min_coin = INT_MAX;
        for (int coin : coins) {
            m[coin] = true;
            min_coin = min(min_coin, coin);
        }

        // 主循环
        vector<int> dp(amount + 1, -1); // 这里尽可能方便表示了
        for (int i = min_coin; i <= amount; i++) {
            // 检查其是否恰好可以用一枚硬币表示
            if (m.find(i) != m.end())
                dp[i] = 1;
            else {
                // 不能恰好用一枚硬币表示，则查看其是否可以由其他状态转移而来
                int min_val = INT_MAX;
                for (int coin : coins) {
                    if (i - coin < 0)
                        continue;
                    if (dp[i - coin] == -1)
                        continue;
                    min_val = min(min_val, dp[i - coin]);
                }
                if (min_val == INT_MAX)
                    dp[i] = -1;
                else
                    dp[i] = min_val + 1;
            }
        }
        return dp[amount];
    }
};
```
如果数组`coin`的长度为`n`，`amount`的值为`m`，那么上面程序的时间复杂度为$O(mn)$，空间复杂度为$O(m)$。
[官方题解](https://leetcode.cn/problems/coin-change/solutions/132979/322-ling-qian-dui-huan-by-leetcode-solution/)的思路也是如此，时间和空间复杂度也是一样的，但是写出的程序要更为紧凑：
```cpp
class Solution {
public:
    int coinChange(vector<int>& coins, int amount) {
        int Max = amount + 1;
        vector<int> dp(amount + 1, Max);
        dp[0] = 0;
        for (int i = 1; i <= amount; ++i) {
            for (int j = 0; j < (int)coins.size(); ++j) {
                if (coins[j] <= i) {
                    dp[i] = min(dp[i], dp[i - coins[j]] + 1);
                }
            }
        }
        return dp[amount] > amount ? -1 : dp[amount];
    }
};
```
官方题解还给出了一种“记忆化搜索”的方式。思路与上面是相似的，只不过作者用了递归这个巧妙的形式，如果对应的解已经存储了，则直接调用对应的值；如果没有存储，则按照问题性质进行递归，并存储对应的值。可以参考：
```cpp
class Solution {
    vector<int>count;
    int dp(vector<int>& coins, int rem) {
        if (rem < 0) return -1;
        if (rem == 0) return 0;
        if (count[rem - 1] != 0) return count[rem - 1];
        int Min = INT_MAX;
        for (int coin:coins) {
            int res = dp(coins, rem - coin);
            if (res >= 0 && res < Min) {
                Min = res + 1;
            }
        }
        count[rem - 1] = Min == INT_MAX ? -1 : Min;
        return count[rem - 1];
    }
public:
    int coinChange(vector<int>& coins, int amount) {
        if (amount < 1) return 0;
        count.resize(amount);
        return dp(coins, amount);
    }
};
```
