在给定的 m x n 网格 grid 中，每个单元格可以有以下三个值之一：

- 值 0 代表空单元格；
- 值 1 代表新鲜橘子；
- 值 2 代表腐烂的橘子。

每分钟，腐烂的橘子 周围 4 个方向上相邻 的新鲜橘子都会腐烂。

返回 直到单元格中没有新鲜橘子为止所必须经过的最小分钟数。如果不可能，返回 -1 。

提示：grid\[i\]\[j] 仅为 0、1 或 2

---
本题很显然是需要使用广度优先遍历，因为只有这种遍历方式才能实现“可控步数”的“腐烂感染”的过程。
在“200. 岛屿数量”的启发之下，本题首先将所有的“橘子”都视为一个上下左右互相连接的无向图。每一分钟（也就是每一次广度优先的迭代），每个腐烂的橘子都会以自己为中心“感染”周围一格的橘子。广度优先会使用队列来存储结点，如果遍历结束时，仍然存在还未被“感染”的橘子，则说明不可能“感染”所有橘子。
很容易写出如下的程序：
```cpp
class Solution {
public:
    int orangesRotting(vector<vector<int>>& grid) {
        int m = grid.size(), n = grid[0].size();

        // 先遍历整个矩阵，获取已经腐烂的橘子和新鲜橘子的数量
        // 并将腐烂橘子提前放入队列，方便之后的广度优先遍历
        queue<pair<int, int>> next_q; // 结点内容为坐标和腐烂时间
        int good = 0;
        for (int i = 0; i < m; i++) {
            for (int j = 0; j < n; j++) {
                if (grid[i][j] == 1)
                    good++;
                else if (grid[i][j] == 2)
                    next_q.push({i, j});
            }
        }
        if (good == 0)
            return 0; // 初始时就没有新鲜的橘子

        // 进行广度优先遍历
        int dx[4] = {-1, 1, 0, 0};
        int dy[4] = {0, 0, -1, 1};
        int counter = -1; // 计时器
        while (!next_q.empty()) {
            queue<pair<int, int>> q;
            swap(q, next_q);

            while (!q.empty()) {
                auto [x, y] = q.front();
                q.pop();

                for (int i = 0; i < 4; i++) {
                    int nx = x + dx[i];
                    int ny = y + dy[i];
                    if (nx < 0 || nx >= m)
                        continue;
                    if (ny < 0 || ny >= n)
                        continue;
                    if (grid[nx][ny] == 1) {
                        next_q.push({nx, ny});
                        grid[nx][ny] = 2;
                        good--;
                    }
                }
            }
            counter++;
        }

        if (good > 0)
            return -1;
        else
            return counter;
    }
};
```
[官方题解](https://leetcode.cn/problems/rotting-oranges/solutions/124765/fu-lan-de-ju-zi-by-leetcode-solution/)的思路与上面的相同，但是因为没有用到两个队列来存储状态，因此运行速度要更快一点：
```cpp
class Solution {
    int cnt;
    int dis[10][10];
    int dir_x[4] = {0, 1, 0, -1};
    int dir_y[4] = {1, 0, -1, 0};
public:
    int orangesRotting(vector<vector<int>>& grid) {
        queue<pair<int, int>>Q;
        memset(dis, -1, sizeof(dis));
        cnt = 0;
        int n = (int)grid.size(), m = (int)grid[0].size(), ans = 0;
        for (int i = 0; i < n; ++i) {
            for (int j = 0; j < m; ++j) {
                if (grid[i][j] == 2) {
                    Q.emplace(i, j);
                    dis[i][j] = 0;
                }
                else if (grid[i][j] == 1) {
                    cnt += 1;
                }
            }
        }
        while (!Q.empty()){
            auto [r, c] = Q.front();
            Q.pop();
            for (int i = 0; i < 4; ++i) {
                int tx = r + dir_x[i];
                int ty = c + dir_y[i];
                if (tx < 0|| tx >= n || ty < 0|| ty >= m || ~dis[tx][ty] || !grid[tx][ty]) {
                    continue;
                }
                dis[tx][ty] = dis[r][c] + 1;
                Q.emplace(tx, ty);
                if (grid[tx][ty] == 1) {
                    cnt -= 1;
                    ans = dis[tx][ty];
                    if (!cnt) {
                        break;
                    }
                }
            }
        }
        return cnt ? -1 : ans;
    }
};
```
因为只需要遍历一次数组和进行一次图的遍历，因此这个算法的时间复杂度为$O(mn)$，空间复杂度也为$O(mn)$。
