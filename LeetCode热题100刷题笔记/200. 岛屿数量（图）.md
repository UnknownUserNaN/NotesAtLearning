给你一个由 '1'（陆地）和 '0'（水）组成的的二维网格，请你计算网格中岛屿的数量。

岛屿总是被水包围，并且每座岛屿只能由水平方向和/或竖直方向上相邻的陆地连接形成。

此外，你可以假设该网格的四条边均被水包围。

---
首先这是一个图的问题，并且这个问题会涉及到图的遍历方式。图的遍历方式有两种，一种是深度优先遍历（DFS），一种是广度优先遍历（BFS）。两者的区别在于扩展节点时使用什么样的数据结构进行存储，使用栈存储（先进后出）就是深度优先遍历，使用队列存储（先进先出）就是广度优先遍历。
这道题我想先用广度优先的思路试一试。思路很简单，设定一个队列，存储待遍历的节点；再设定一个与网格大小同形的数组，用来存放该节点是否遍历过（-1为未遍历），是否是水域（0），岛屿编号（1及以上）。最开始的时候将图中随便的一个结点放入队列，然后一直循环，扩展结点、设定结点状态、根据周围结点设定标号，直到队列为空。
最开始写出的程序是这样的：
```cpp
class Solution {
public:
    int numIslands(vector<vector<char>>& grid) {
        int m = grid.size(), n = grid[0].size();
        vector<vector<int>> visited(m, vector<int>(n, -1));
        queue<pair<int, int>> q;
        int dx[4] = {1, -1, 0, 0};
        int dy[4] = {0, 0, 1, -1};
        int counter = 0;

        q.push({0, 0});
        while (!q.empty()) {
            // 检查周围节点状态，靠周围节点取值确定当前节点状态
            auto [x, y] = q.front();
            q.pop();
            for (int i = 0; i < 4; i++) {
                int nx = x + dx[i];
                int ny = y + dy[i];
                if (nx < 0 || nx >= m)
                    continue;
                if (ny < 0 || ny >= n)
                    continue;
                if (visited[nx][ny] == -1)
                    q.push({nx, ny});
                if (grid[x][y] != '0') {
                    if (grid[nx][ny] != '0' && visited[nx][ny] > 0) {
                        visited[x][y] = visited[nx][ny];
                        break;
                    }
                } else
                    visited[x][y] = 0;
            }
            if (grid[x][y] != '0' && visited[x][y] == -1) {
                counter++;
                visited[x][y] = counter;
            }
        }
        return counter;
    }
};
```
但很遗憾的是，这个程序在运行下面的用例时出问题了：
```text
[["1","1","1"],
 ["0","1","0"],
 ["1","1","1"]]
```
这个用例是一个“工”字型的“岛屿”，返回值应为1，但是上面的程序返回的是1。因为上面的程序在遍历到左下角的结点时，会为其赋为`2`。这主要是因为上面的程序在设计时假设在遍历时，每个作为“陆地”的结点都会相邻，因此相互联通的陆地自然可以被赋为一样的岛屿编号。但事实上不是这样的。
遗憾的是，为了解决这个问题，我没有想出什么比较好的方法。最终只能去诉诸[官方题解](https://leetcode.cn/problems/number-of-islands/solutions/13103/dao-yu-shu-liang-by-leetcode/)。
官方题解有两点非常醍醐灌顶：
1. 将图中为陆地的结点都视为互相连接的有向图，因此题中的每个岛屿都可以被视为一个无向图，多个岛屿对应着多个不互相联通的无向图。
2. 我们遍历整个`grid`矩阵找到所有没有访问过的陆地结点，然后从该结点开始进行遍历，遍历完该结点所属岛屿的所有结点，并将相应的结点确定访问过。（不过官方题解这里直接修改了原数组，不能修改原数组的话就得加入标记数组）
基于这两点，就很容易设计出对应的算法了。下面直接贴上官方题解的做法。首先是深度遍历：
```cpp
class Solution {
private:
    void dfs(vector<vector<char>>& grid, int r, int c) {
        int nr = grid.size();
        int nc = grid[0].size();

        grid[r][c] = '0';
        if (r - 1 >= 0 && grid[r-1][c] == '1') dfs(grid, r - 1, c);
        if (r + 1 < nr && grid[r+1][c] == '1') dfs(grid, r + 1, c);
        if (c - 1 >= 0 && grid[r][c-1] == '1') dfs(grid, r, c - 1);
        if (c + 1 < nc && grid[r][c+1] == '1') dfs(grid, r, c + 1);
    }

public:
    int numIslands(vector<vector<char>>& grid) {
        int nr = grid.size();
        if (!nr) return 0;
        int nc = grid[0].size();

        int num_islands = 0;
        for (int r = 0; r < nr; ++r) {
            for (int c = 0; c < nc; ++c) {
                if (grid[r][c] == '1') {
                    ++num_islands;
                    dfs(grid, r, c);
                }
            }
        }

        return num_islands;
    }
};
```
然后是广度遍历：
```cpp
class Solution {
public:
    int numIslands(vector<vector<char>>& grid) {
        int nr = grid.size();
        if (!nr) return 0;
        int nc = grid[0].size();

        int num_islands = 0;
        for (int r = 0; r < nr; ++r) {
            for (int c = 0; c < nc; ++c) {
                if (grid[r][c] == '1') {
                    ++num_islands;
                    grid[r][c] = '0';
                    queue<pair<int, int>> neighbors;
                    neighbors.push({r, c});
                    while (!neighbors.empty()) {
                        auto rc = neighbors.front();
                        neighbors.pop();
                        int row = rc.first, col = rc.second;
                        if (row - 1 >= 0 && grid[row-1][col] == '1') {
                            neighbors.push({row-1, col});
                            grid[row-1][col] = '0';
                        }
                        if (row + 1 < nr && grid[row+1][col] == '1') {
                            neighbors.push({row+1, col});
                            grid[row+1][col] = '0';
                        }
                        if (col - 1 >= 0 && grid[row][col-1] == '1') {
                            neighbors.push({row, col-1});
                            grid[row][col-1] = '0';
                        }
                        if (col + 1 < nc && grid[row][col+1] == '1') {
                            neighbors.push({row, col+1});
                            grid[row][col+1] = '0';
                        }
                    }
                }
            }
        }

        return num_islands;
    }
};
```
程序的实现方式大同小异，但核心是不变的。最终的时间复杂度为$O(mn)$，空间复杂度也为$O(mn)$。
值得一提的是，官方题解最后还提到了一种“并查集”的做法。但由于该做法的时间复杂度和空间复杂度较上面没有明显改善，这里便不再赘述。
