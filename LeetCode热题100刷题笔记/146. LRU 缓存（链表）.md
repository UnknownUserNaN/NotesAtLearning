请你设计并实现一个满足  LRU (最近最少使用) 缓存 约束的数据结构。
实现 LRUCache 类：
1. LRUCache(int capacity) 以 正整数 作为容量 capacity 初始化 LRU 缓存
2. int get(int key) 如果关键字 key 存在于缓存中，则返回关键字的值，否则返回 -1 。
3. void put(int key, int value) 如果关键字 key 已经存在，则变更其数据值 value ；如果不存在，则向缓存中插入该组 key-value 。如果插入操作导致关键字数量超过 capacity ，则应该 逐出 最久未使用的关键字。
函数 get 和 put 必须以 O(1) 的平均时间复杂度运行。

---
想法很简单，就是使用链表作为每个LRU单元的基础存储，这样便容易实现增删的操作。
但是根据题目要求，我们还需要实现常数时间的增删改操作，如果只使用链表这是非常难以实现的。这样的要求让我们想到了哈希表，因此我们还需要建立一个哈希表，实现键值到对应链表的一一映射。此外，为了更好地实现记录最久未使用的关键字的操作，我们还需要建立双向链表，以实现$O(1)$时间的增删改操作。
所以，综合起来，题目要求的函数主要执行以下的几个操作：
1. LRUCache(int capacity): 储存缓存容量；确定链表的空的头结点；建立一个空的哈希表；初始化当前缓存的长度。
2. void put(int key, int value): 先根据输入键值查询哈希表，如果哈希表中有对应元素，则修改对应链表节点的键值，并将对应单元移动到链表末尾；否则，增加一个新结点连接到链表中，并在哈希表中增加对应元素。如果这一增加长度的操作使得链表长度超出了缓存长度限制，那么链表的头指针向后移动一位，并释放掉最开头的链表节点的内存。
3. int get(int key): 根据键值查询哈希表中是否存在对应元素，若有则返回对应链表单元的值，并将该单元移动到链表末尾，若无则返回-1。
基于这样的想法，很容易写出如下的程序。但是下面的程序不知为何会在运行某些样例时报错：
```cpp
class LRUCache {
private:
    struct ListNode {
        int val;
        ListNode* prev;
        ListNode* next;
        ListNode(int x) : val(x), prev(nullptr), next(nullptr) {}
    };
    int length;
    int capacity;
    ListNode *head, *tail;
    unordered_map<int, ListNode*> dict;

public:
    LRUCache(int capacity) {
        this->capacity = capacity;
        this->length = 0;
        head = nullptr;
        tail = nullptr;
    }

    void put(int key, int value) {
        // 先查询哈希表中是否有对应的项
        if (dict.find(key) != dict.end()) {
            // 存在对应的项
            ListNode* p = dict[key];
            p->val = value;
            // 将该项移动到链表末尾
            if (this->length != 1) {
                if (p == head) {
                    // 要移动的链表是头结点
                    head = p->next;
                    head->prev = nullptr;
                    tail->next = p;
                    p->prev = tail;
                    p->next = nullptr;
                    tail = p;
                } else if (p != tail) {
                    // 要移动的链表不是头结点，也不是尾结点
                    p->prev->next = p->next;
                    p->next->prev = p->prev;
                    tail->next = p;
                    p->prev = tail;
                    p->next = nullptr;
                    tail = p;
                }
            }
        } else {
            // 不存在对应的项，则增加对应的项
            ListNode* p = new ListNode(value);
            if (head == nullptr) {
                head = p;
                tail = p;
                p->next = nullptr;
                p->prev = nullptr;
            } else {
                tail->next = p;
                p->prev = tail;
                p->next = nullptr;
                tail = p;
            }
            dict[key] = p;
            this->length++;
            if (this->length > this->capacity) {
                // 删除最久远的元素，也就是链表的头结点
                dict.erase(head->val);
                ListNode* temp = head->next;
                delete head;
                head = temp;
                this->length--;
            }
        }
    }

    int get(int key) {
        if (dict.find(key) != dict.end()) {
            // 将该项移动到链表末尾
            ListNode* p = dict[key];
            if (this->length != 1) {
                if (p == head) {
                    // 要移动的链表是头结点
                    head = p->next;
                    head->prev = nullptr;
                    tail->next = p;
                    p->prev = tail;
                    p->next = nullptr;
                    tail = p;
                } else if (p != tail) {
                    // 要移动的链表不是头结点，也不是尾结点
                    p->prev->next = p->next;
                    p->next->prev = p->prev;
                    tail->next = p;
                    p->prev = tail;
                    p->next = nullptr;
                    tail = p;
                }
            }
            return dict[key]->val;
        } else
            return -1;
    }
};
```

这里。对于两个只是修改了输入值的用例，偏偏就是无法通过。
官方题解中的思路与上面大同小异，区别在于官方题解给题中的双向链表增加了head和tail两个空结点，这样在进行增删改时无需判断当前节点是否是头、尾结点，省去了额外的特殊处理。不得不说这是一个非常有用的技巧：
```cpp
struct DLinkedNode {
    int key, value;
    DLinkedNode* prev;
    DLinkedNode* next;
    DLinkedNode(): key(0), value(0), prev(nullptr), next(nullptr) {}
    DLinkedNode(int _key, int _value): key(_key), value(_value), prev(nullptr), next(nullptr) {}
};

class LRUCache {
private:
    unordered_map<int, DLinkedNode*> cache;
    DLinkedNode* head;
    DLinkedNode* tail;
    int size;
    int capacity;

public:
    LRUCache(int _capacity): capacity(_capacity), size(0) {
        // 使用伪头部和伪尾部节点
        head = new DLinkedNode();
        tail = new DLinkedNode();
        head->next = tail;
        tail->prev = head;
    }
    
    int get(int key) {
        if (!cache.count(key)) {
            return -1;
        }
        // 如果 key 存在，先通过哈希表定位，再移到头部
        DLinkedNode* node = cache[key];
        moveToHead(node);
        return node->value;
    }
    
    void put(int key, int value) {
        if (!cache.count(key)) {
            // 如果 key 不存在，创建一个新的节点
            DLinkedNode* node = new DLinkedNode(key, value);
            // 添加进哈希表
            cache[key] = node;
            // 添加至双向链表的头部
            addToHead(node);
            ++size;
            if (size > capacity) {
                // 如果超出容量，删除双向链表的尾部节点
                DLinkedNode* removed = removeTail();
                // 删除哈希表中对应的项
                cache.erase(removed->key);
                // 防止内存泄漏
                delete removed;
                --size;
            }
        }
        else {
            // 如果 key 存在，先通过哈希表定位，再修改 value，并移到头部
            DLinkedNode* node = cache[key];
            node->value = value;
            moveToHead(node);
        }
    }

    void addToHead(DLinkedNode* node) {
        node->prev = head;
        node->next = head->next;
        head->next->prev = node;
        head->next = node;
    }
    
    void removeNode(DLinkedNode* node) {
        node->prev->next = node->next;
        node->next->prev = node->prev;
    }

    void moveToHead(DLinkedNode* node) {
        removeNode(node);
        addToHead(node);
    }

    DLinkedNode* removeTail() {
        DLinkedNode* node = tail->prev;
        removeNode(node);
        return node;
    }
};
```
