给你一个字符串 s，请你将 s 分割成一些 子串，使每个子串都是 回文串 。返回 s 所有可能的分割方案。

---
先记录一个我见到这个问题时的直觉，这个问题很可能有效率更高的做法，而且很可能不是回溯法。但无论如何，回溯法解决问题是最简单的。
回溯法内涵着递归的思想。当我们“拿到”一个字符串，我们便从头开始遍历寻找回文串。如果当前遍历的部分构成了回文串，我们便立刻将子串放入结果数组，然后将字符串剩余部分的划分交给回溯来完成。
大体的思路是这样，但这里还有两个可以优化的点。一是寻找回文串的算法，如果我们每移动一位就判断子串是否“对称”，那这样做的时间复杂度显然比较高，因为这样做的时间复杂度是$O(n)$。可能存在着一种$O(1)$的算法能解决这个问题。二是可以使用缓存来存储回溯过程中遇到的各种子串，这相当于是一种空间换时间的做法。但可惜这么做有些麻烦，对于练习用的算法题来说，这样还是太“奢侈”了点。
总之我们先按最简单的实现方法完成本题：
```cpp
class Solution {
public:
    bool isvalid(string s, int l, int r) {
        for (int i = l; i <= (l + r) / 2; i++)
            if (s[i] != s[l + r - i])
                return false;
        return true;
    }

    void backtracking(vector<vector<string>>& ret, vector<string>& com,
                      string s, int cur) {
        if (cur > s.size() - 1) {
            ret.push_back(com);
            return;
        }

        for (int i = cur; i < s.size(); i++)
            if (isvalid(s, cur, i)) {
                com.push_back(s.substr(cur, i - cur + 1));
                backtracking(ret, com, s, i + 1);
                com.pop_back();
            }
    }

    vector<vector<string>> partition(string s) {
        vector<vector<string>> ret;
        vector<string> com;
        backtracking(ret, com, s, 0);
        return ret;
    }
};
```
最终得到的程序用时75ms，击败了55.21%的人。说明程序还是有优化空间，那就不妨按照之前的思路进行优化。最需要优化的就是判断回文串的方法。
考虑字符串的在闭区间$[i,j]$上的子串，直接判断其是否是回文串需要调用上面$O(n)$的算法。如果不通过该方法判断，那就考虑划归的思路，将其拆分成一个规模更小的子问题。想到这里思路就比较清晰了，如果在$[i,j]$上是否是子串，与在$[i+1, j-1]$上是否是子串有密切关系。如果在$[i+1, j-1]$上是子串，那么如果我们又知道了$s[i]=s[j]$，便可以直接确定在在$[i,j]$上是子串。这实际上是一个动态规划的做法，我们可以直接写出动态规划的状态转移方程：
$$dp[i][j]=\left\{
\begin{aligned}
true&,j\leq i\\
dp[i+1][j-1]\&\&\left(s[i]==s[j]\right)&,j>i
\end{aligned}
\right.$$
建立了上面的动态规划数组之后，我们便可以在$O(1)$的时间内确知子串是否是回文串了。当然这是一种“空间换时间”的做法，这样的代价是我们要建立一个大小为$n^2$的矩阵（当然这个矩阵实际上只有半个三角区域有用，这也是潜在的可以降低占用空间的优化点）。
基于此就可以写出更新后的代码：
```cpp
class Solution {
public:
    vector<vector<bool>> build(string& s) {
        int n = s.size();
        vector<vector<bool>> dp(n, vector<bool>(n, true));
        for (int j = 0; j < n; j++)
            for (int i = 0; i < j; i++)
                dp[i][j] = dp[i + 1][j - 1] && (s[i] == s[j]);
        return dp;
    }

    bool isvalid(vector<vector<bool>>& dp, int l, int r) { return dp[l][r]; }

    void backtracking(vector<vector<string>>& ret, vector<string>& com,
                      vector<vector<bool>>& dp, string s, int cur) {
        if (cur > s.size() - 1) {
            ret.push_back(com);
            return;
        }

        for (int i = cur; i < s.size(); i++)
            if (isvalid(dp, cur, i)) {
                com.push_back(s.substr(cur, i - cur + 1));
                backtracking(ret, com, dp, s, i + 1);
                com.pop_back();
            }
    }

    vector<vector<string>> partition(string s) {
        auto dp = build(s);
        vector<vector<string>> ret;
        vector<string> com;
        backtracking(ret, com, dp, s, 0);
        return ret;
    }
};
```
[官方题解](https://leetcode.cn/problems/palindrome-partitioning/solutions/639633/fen-ge-hui-wen-chuan-by-leetcode-solutio-6jkv/)的写法与我的做法大同小异，但是官方题解在细节处的实现更为巧妙。首先将动态规划的数组定义成全局变量，由于全局变量定义时并不知道输入字符串的长度，所以后面转而在partition函数中使用assign方法对其进行初始化。
```cpp
class Solution {
private:
    vector<vector<int>> f;
    vector<vector<string>> ret;
    vector<string> ans;
    int n;

public:
    void dfs(const string& s, int i) {
        if (i == n) {
            ret.push_back(ans);
            return;
        }
        for (int j = i; j < n; ++j) {
            if (f[i][j]) {
                ans.push_back(s.substr(i, j - i + 1));
                dfs(s, j + 1);
                ans.pop_back();
            }
        }
    }

    vector<vector<string>> partition(string s) {
        n = s.size();
        f.assign(n, vector<int>(n, true));

        for (int i = n - 1; i >= 0; --i) {
            for (int j = i + 1; j < n; ++j) {
                f[i][j] = (s[i] == s[j]) && f[i + 1][j - 1];
            }
        }

        dfs(s, 0);
        return ret;
    }
};
```
该回溯方法中，每个字符都有两种状态，即“进入答案数组”或“不进入答案数组”，因此整个“回溯树”有$2^n$个结点，每个结点处要进行$O(n)$的尝试，所以总的时间复杂度是$O(n\cdot 2^n)$。空间复杂度是$O(n^2)$（因为使用了动态规划数组）。
当然，官方题解还提供了“一边回溯、一边建立动态规划数组”的优化方法。这里的程序设计方法非常值得学习：即若答案已有储存，则直接返回答案；否则则按照递推的方法寻找答案，并存储答案。这样写出的程序时间复杂度和空间复杂度与上相同，但是实际运行时间要比上面的做法略低。
```cpp
class Solution {
private:
    vector<vector<int>> f;
    vector<vector<string>> ret;
    vector<string> ans;
    int n;

public:
    void dfs(const string& s, int i) {
        if (i == n) {
            ret.push_back(ans);
            return;
        }
        for (int j = i; j < n; ++j) {
            if (isPalindrome(s, i, j) == 1) {
                ans.push_back(s.substr(i, j - i + 1));
                dfs(s, j + 1);
                ans.pop_back();
            }
        }
    }

    // 记忆化搜索中，f[i][j] = 0 表示未搜索，1 表示是回文串，-1 表示不是回文串
    int isPalindrome(const string& s, int i, int j) {
        if (f[i][j]) {
            return f[i][j];
        }
        if (i >= j) {
            return f[i][j] = 1;
        }
        return f[i][j] = (s[i] == s[j] ? isPalindrome(s, i + 1, j - 1) : -1);
    }

    vector<vector<string>> partition(string s) {
        n = s.size();
        f.assign(n, vector<int>(n));

        dfs(s, 0);
        return ret;
    }
};
```
