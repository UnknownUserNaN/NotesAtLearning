Trie（发音类似 "try"）或者说 前缀树 是一种树形数据结构，用于高效地存储和检索字符串数据集中的键。这一数据结构有相当多的应用情景，例如自动补全和拼写检查。

请你实现 Trie 类：

- Trie() 初始化前缀树对象。
- void insert(String word) 向前缀树中插入字符串 word 。
- boolean search(String word) 如果字符串 word 在前缀树中，返回 true（即，在检索之前已经插入）；否则，返回 false 。
- boolean startsWith(String prefix) 如果之前已经插入的字符串 word 的前缀之一为 prefix ，返回 true ；否则，返回 false 。

提示：word 和 prefix 仅由小写英文字母组成

---
一个非常简单的思路是，建立一棵树，树的每一个结点存储一个字母。当我们从树的根开始逐步向叶子遍历时，这个路径就可以按顺序组成一个字符串。不过，为了表示一个单词结束，我们还需要在插入完单词中各个字母的结点后再插入一个结束符的结点。
这就是我们最基本的思路，基于此很容易写出对应的程序：
```cpp
class Trie {
private:
    // 自定义树节点
    struct TreeNode {
        char c;
        unordered_map<char, TreeNode*> children;
        TreeNode() : c('\0'), children({}) {}
        TreeNode(char ch) : c(ch), children({}) {}
    };
    TreeNode* root;

public:
    Trie() { root = new TreeNode(); }

    void insert(string word) {
        // 逐个字母地创建节点
        TreeNode* p = root;
        for (char ch : word) {
            // 寻找是否存在该节点
            auto ret = p->children.find(ch);
            if (ret != p->children.end())
                // 存在该节点
                p = (*ret).second;
            else {
                // 不存在该节点，则创建该节点
                TreeNode* new_node = new TreeNode(ch);
                p->children[ch] = new_node;
                p = p->children[ch];
            }
        }
        // 为了表示单词已结束，再插入一个空结点作为结束符
        TreeNode* end_node = new TreeNode('0');
        p->children['0'] = end_node;
    }

    bool search(string word) {
        TreeNode* p = root;
        for (char ch : word) {
            auto ret = p->children.find(ch);
            if (ret == p->children.end())
                return false;
            p = (*ret).second;
        }
        return p->children.find('0') != p->children.end();
    }

    bool startsWith(string prefix) {
        TreeNode* p = root;
        for (char ch : prefix) {
            auto ret = p->children.find(ch);
            if (ret == p->children.end())
                return false;
            p = (*ret).second;
        }
        return true;
    }
};

 * Trie* obj = new Trie();
 * obj->insert(word);
 * bool param_2 = obj->search(word);
 * bool param_3 = obj->startsWith(prefix);
 */
```
程序很顺利地AC了，但是很显然，上面程序的运行时间还可以优化。
[官方题解](https://leetcode.cn/problems/implement-trie-prefix-tree/solutions/717239/shi-xian-trie-qian-zhui-shu-by-leetcode-ti500/)则用递归的结构实现了上面的思路：
```cpp
class Trie {
private:
    vector<Trie*> children;
    bool isEnd;

    Trie* searchPrefix(string prefix) {
        Trie* node = this;
        for (char ch : prefix) {
            ch -= 'a';
            if (node->children[ch] == nullptr) {
                return nullptr;
            }
            node = node->children[ch];
        }
        return node;
    }

public:
    Trie() : children(26), isEnd(false) {}

    void insert(string word) {
        Trie* node = this;
        for (char ch : word) {
            ch -= 'a';
            if (node->children[ch] == nullptr) {
                node->children[ch] = new Trie();
            }
            node = node->children[ch];
        }
        node->isEnd = true;
    }

    bool search(string word) {
        Trie* node = this->searchPrefix(word);
        return node != nullptr && node->isEnd;
    }

    bool startsWith(string prefix) {
        return this->searchPrefix(prefix) != nullptr;
    }
};
```
评论区有人给出了智能指针的实现版本：
```cpp
class Trie {
private:
    struct Node {
        bool isEnd;
        vector<unique_ptr<Node>> children;

        Node() : isEnd(false), children(26) {}
    };

    unique_ptr<Node> head;

    Node* searchTail(string& word) {
        auto cur = head.get();
        for (const char& ch : word) {
            int idx = ch - 'a';
            if (!cur->children[idx]) {
                return nullptr;
            }
            cur = cur->children[idx].get();
        }
        return cur;
    }

public:
    Trie() { head = make_unique<Node>(); }

    void insert(string word) {
        auto cur = head.get();
        for (const char& ch : word) {
            int idx = ch - 'a';
            if (!cur->children[idx]) {
                cur->children[idx] = make_unique<Node>();
            }
            cur = cur->children[idx].get();
        }
        cur->isEnd = true;
    }

    bool search(string word) {
        auto tail = searchTail(word);
        return tail ? tail->isEnd : false;
    }

    bool startsWith(string prefix) {
        auto tail = searchTail(prefix);
        return tail != nullptr;
    }
};
```
