按照国际象棋的规则，皇后可以攻击与之处在同一行或同一列或同一斜线上的棋子。

n 皇后问题 研究的是如何将 n 个皇后放置在 n×n 的棋盘上，并且使皇后彼此之间不能相互攻击。

给你一个整数 n ，返回所有不同的 n 皇后问题 的解决方案。

每一种解法包含一个不同的 n 皇后问题 的棋子放置方案，该方案中 'Q' 和 '.' 分别代表了皇后和空位。

---
这是一个非常经典的回溯问题。
使用回溯法，本题很容易得到思路。对于每一时刻的回溯，我们枚举棋盘中所有可能的位置，然后检测该位置是否与已经放置的棋子在同一横行或者纵列，以及是否在同一斜线上。如果该位置通过检查，则将其加入数组，然后进行下一步的回溯。
第一次写出的程序如下：
```cpp
class Solution {
public:
    int abs(int x) { return x < 0 ? -x : x; }

    void backtracking(vector<vector<string>>& ret, vector<string>& board,
                      vector<int>& pos_x, vector<int>& pos_y, int n) {
        if (pos_x.size() == n) {
            ret.push_back(board);
            return;
        }

        for (int i = 0; i < n; i++) {
            // 检查该横行是否已经被“占据”
            if (find(pos_x.begin(), pos_x.end(), i) != pos_x.end())
                continue;

            for (int j = 0; j < n; j++) {
                // 检查纵列是否已经被“占据”
                if (find(pos_y.begin(), pos_y.end(), j) != pos_y.end())
                    continue;

                // 检测该位置是否与棋子在同一条斜线上
                bool is_valid = true;
                for (int k = 0; k < pos_x.size(); k++)
                    if (abs(i - pos_x[k]) == abs(j - pos_y[k])) {
                        is_valid = false;
                        break;
                    }
                if (!is_valid)
                    continue;

                // 该位置可以加入解集
                pos_x.push_back(i);
                pos_y.push_back(j);
                board[i][j] = 'Q';
                backtracking(ret, board, pos_x, pos_y, n);
                pos_x.pop_back();
                pos_y.pop_back();
                board[i][j] = '.';
            }
        }
    }

    vector<vector<string>> solveNQueens(int n) {
        vector<vector<string>> ret;
        vector<string> board;
        vector<int> pos_x, pos_y;

        string s;
        for (int i = 0; i < n; i++)
            s += ".";
        board.assign(n, s);

        backtracking(ret, board, pos_x, pos_y, n);
        return ret;
    }
};
```
但是很遗憾，这个程序无法通过`N=4`时的检查。根据题目用例，`N=4`时的合理答案只有2种，但是上面的程序却输出了48种结果，其中绝大部分的结果是重复的。我们当然可以将上面提取出的结果进行去重，从而得到正确的结果，但是这样显然会降低很多的效率，程序必然会运行超时。
问题出在哪里呢？在D老师的帮助下，我发现问题出在上面的代码里，把最终结果的“组合”问题当做了“排列”。`N=4`时有4个棋子，2种组合方式，但上面的程序列举了这4个棋子所有合理的“排列”，恰好是$A_4^4=24$种，两种组合方式，于是就得到了最终答案里的$2\times24=48$种。
> 你的算法在逻辑上看似“全盘搜索”了所有不同行、列、对角线都不冲突的皇后摆放，但实际上它会把同一个最终棋盘解，按照不同的“放置顺序”重复记录多次。以 n=4 为例，正确的解只有 2 个，但你的程序会返回 48 个（每个解被 24 种不同的放置顺序重复添加）——LeetCode 自然判定你的答案错误。

问题找到了，接下来要想办法把问题处理成组合问题，我们要保证列举不再“重复”。
D老师给出了一种最简单也最高效的改法。由于题中$n\times n$的棋盘上要放$n$个棋子，那么很显然棋子必定会占据所有的行和列，也就是说，当我们按行或列去遍历棋盘的时候，该行或该列必然只有一个棋子。那么我们可以以此来做回溯，当结果数组中有$i$个元素时，我们只考虑棋子在$i$行的位置情况，而不是在整个棋盘上枚举它可能的位置。
基于此可以很快写出改进后的代码：
```cpp
class Solution {
public:
    int abs(int x) { return x < 0 ? -x : x; }

    void backtracking(vector<vector<string>>& ret, vector<string>& board,
                      vector<int>& pos_x, vector<int>& pos_y, int n) {
        if (pos_x.size() == n) {
            if (ret.size() != 0)
                for (auto t : ret)
                    if (t == board)
                        return;
            ret.push_back(board);
            return;
        }

        int i = pos_x.size();
        for (int j = 0; j < n; j++) {
            // 检查纵列是否已经被“占据”
            if (find(pos_y.begin(), pos_y.end(), j) != pos_y.end())
                continue;

            // 检测该位置是否与棋子在同一条斜线上
            bool is_valid = true;
            for (int k = 0; k < pos_x.size(); k++)
                if (abs(i - pos_x[k]) == abs(j - pos_y[k])) {
                    is_valid = false;
                    break;
                }
            if (!is_valid)
                continue;

            // 该位置可以加入解集
            pos_x.push_back(i);
            pos_y.push_back(j);
            board[i][j] = 'Q';
            backtracking(ret, board, pos_x, pos_y, n);
            pos_x.pop_back();
            pos_y.pop_back();
            board[i][j] = '.';
        }
    }

    vector<vector<string>> solveNQueens(int n) {
        vector<vector<string>> ret;
        vector<string> board;
        vector<int> pos_x, pos_y;

        string s;
        for (int i = 0; i < n; i++)
            s += ".";
        board.assign(n, s);

        backtracking(ret, board, pos_x, pos_y, n);
        return ret;
    }
};
```
这次是直接AC了，但是程序的用时略高，用时39ms，仅超越10.37%的人。
[官方题解](https://leetcode.cn/problems/n-queens/solutions/398929/nhuang-hou-by-leetcode-solution/)的做法更好一些，使用了额外的空间记录了斜线上的状态，从而省去了遍历和计算的成本。
```cpp
class Solution {
public:
    vector<vector<string>> solveNQueens(int n) {
        auto solutions = vector<vector<string>>();
        auto queens = vector<int>(n, -1);
        auto columns = unordered_set<int>();
        auto diagonals1 = unordered_set<int>();
        auto diagonals2 = unordered_set<int>();
        backtrack(solutions, queens, n, 0, columns, diagonals1, diagonals2);
        return solutions;
    }

    void backtrack(vector<vector<string>> &solutions, vector<int> &queens, int n, int row, unordered_set<int> &columns, unordered_set<int> &diagonals1, unordered_set<int> &diagonals2) {
        if (row == n) {
            vector<string> board = generateBoard(queens, n);
            solutions.push_back(board);
        } else {
            for (int i = 0; i < n; i++) {
                if (columns.find(i) != columns.end()) {
                    continue;
                }
                int diagonal1 = row - i;
                if (diagonals1.find(diagonal1) != diagonals1.end()) {
                    continue;
                }
                int diagonal2 = row + i;
                if (diagonals2.find(diagonal2) != diagonals2.end()) {
                    continue;
                }
                queens[row] = i;
                columns.insert(i);
                diagonals1.insert(diagonal1);
                diagonals2.insert(diagonal2);
                backtrack(solutions, queens, n, row + 1, columns, diagonals1, diagonals2);
                queens[row] = -1;
                columns.erase(i);
                diagonals1.erase(diagonal1);
                diagonals2.erase(diagonal2);
            }
        }
    }

    vector<string> generateBoard(vector<int> &queens, int n) {
        auto board = vector<string>();
        for (int i = 0; i < n; i++) {
            string row = string(n, '.');
            row[queens[i]] = 'Q';
            board.push_back(row);
        }
        return board;
    }
};
```
该算法的空间复杂度容易得出，是$O(n)$。时间复杂度为$O(n!)$，这是因为遍历第$i$个棋子时，其有$n-i$个合理的位置，所以最终的“回溯树”有$n(n-1)(n-2)...2\cdot1$个结点。
官方题解还给出了一种基于二进制移位运算的做法，能节省许多额外空间，但是整体的空间复杂度仍然会维持在$O(n)$的水平。
```cpp
class Solution {
public:
    vector<vector<string>> solveNQueens(int n) {
        auto solutions = vector<vector<string>>();
        auto queens = vector<int>(n, -1);
        solve(solutions, queens, n, 0, 0, 0, 0);
        return solutions;
    }

    void solve(vector<vector<string>> &solutions, vector<int> &queens, int n, int row, int columns, int diagonals1, int diagonals2) {
        if (row == n) {
            auto board = generateBoard(queens, n);
            solutions.push_back(board);
        } else {
            int availablePositions = ((1 << n) - 1) & (~(columns | diagonals1 | diagonals2));
            while (availablePositions != 0) {
                int position = availablePositions & (-availablePositions);
                availablePositions = availablePositions & (availablePositions - 1);
                int column = __builtin_ctz(position);
                queens[row] = column;
                solve(solutions, queens, n, row + 1, columns | position, (diagonals1 | position) >> 1, (diagonals2 | position) << 1);
                queens[row] = -1;
            }
        }
    }

    vector<string> generateBoard(vector<int> &queens, int n) {
        auto board = vector<string>();
        for (int i = 0; i < n; i++) {
            string row = string(n, '.');
            row[queens[i]] = 'Q';
            board.push_back(row);
        }
        return board;
    }
};
```
