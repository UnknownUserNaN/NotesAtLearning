给你两个 非空 的链表，表示两个非负的整数。它们每位数字都是按照 逆序 的方式存储的，并且每个节点只能存储 一位 数字。

请你将两个数相加，并以相同形式返回一个表示和的链表。

你可以假设除了数字 0 之外，这两个数都不会以 0 开头。

示例：
输入：l1 = [2,4,3], l2 = [5,6,4]
输出：[7,0,8]
解释：342 + 465 = 807.

---
题目给出的倒序形式非常“贴心”，因为这模拟了我们进行多位数加法运算时的逐位相加的操作。同时，这还保证了输入的两个数字均从个位“对齐”，更是方便了我们的操作。
接下来的思路就非常简单，只需要依次同步遍历两个链表，然后模拟多位数加法的操作即可。程序也非常简单：
```cpp
/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode() : val(0), next(nullptr) {}
 *     ListNode(int x) : val(x), next(nullptr) {}
 *     ListNode(int x, ListNode *next) : val(x), next(next) {}
 * };
 */
class Solution {
public:
    ListNode* addTwoNumbers(ListNode* l1, ListNode* l2) {
        ListNode *p1 = l1, *p2 = l2;
        ListNode* head = new ListNode(0); // 头节点，什么东西也不存储
        ListNode* p = head;
        int flag = 0; // 进一的标志位
        while (p1 != nullptr || p2 != nullptr) {
            int val1, val2;
            val1 = p1 == nullptr ? 0 : p1->val;
            val2 = p2 == nullptr ? 0 : p2->val;

            int sum = val1 + val2 + flag;
            flag = sum / 10;
            p->next = new ListNode(sum % 10);
            p = p->next;
            p1 = p1 == nullptr ? nullptr : p1->next;
            p2 = p2 == nullptr ? nullptr : p2->next;
        }
        if (flag != 0) {
            p->next = new ListNode(flag);
        }
        return head->next;
    }
};
```
这与官方题解的做法基本一致：
```cpp
class Solution {
public:
    ListNode* addTwoNumbers(ListNode* l1, ListNode* l2) {
        ListNode *head = nullptr, *tail = nullptr;
        int carry = 0;
        while (l1 || l2) {
            int n1 = l1 ? l1->val: 0;
            int n2 = l2 ? l2->val: 0;
            int sum = n1 + n2 + carry;
            if (!head) {
                head = tail = new ListNode(sum % 10);
            } else {
                tail->next = new ListNode(sum % 10);
                tail = tail->next;
            }
            carry = sum / 10;
            if (l1) {
                l1 = l1->next;
            }
            if (l2) {
                l2 = l2->next;
            }
        }
        if (carry > 0) {
            tail->next = new ListNode(carry);
        }
        return head;
    }
};
```
